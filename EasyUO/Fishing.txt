goto Move
}

; ====================FISH====================
Fish:


Set %Chance %Chance + 1
if %Chance > %Tentativa
{
goto loop
}

gosub pole

if #weight > 300
{
gosub Guardar
}

Set #SCNT 0

}

{
msg Raise Anchor$
wait 2s

}

{
msg Raise Anchor$
wait 2s

}

IF %ComandoX > 6
{
Set %ComandoX 1
Set %Comando %Comando + 1

IF %Comando > 4
{
Set %Comando 1
}
}

wait 6s
msg Drop Anchor$
goto loop

sub pole
finditem %Pole
if #findkind = 0
{
set #LObjectID #FindID
event macro 17 0
target
event macro 25 0
event macro 22 0
return
}

if #findkind = 1
{
ignoreitem #FindID
gosub pole
return
}

if #findkind = -1
{
ignoreitem reset
event sysmessage Pole nao encontrado
wait 3s
return
}

sub Guardar

finditem %Peixe C_ , #BackPackID
{
repeat
finditem %Peixe C_ , #BACKPACKID
{
exevent drag #findid #findstack
wait 2s
exevent dropc %id_bau
ignoreitem #findid
}
until #weight =< 35   ;coloque o peso inicial ou que você gostaria que ele retornasse o macro
}

return
